### Database Schema Additions

```sql
-- New category enum expansion
ALTER TYPE category ADD VALUE 'Fitness';
ALTER TYPE category ADD VALUE 'Driving';
ALTER TYPE category ADD VALUE 'PetTraining';
ALTER TYPE category ADD VALUE 'Creative';
ALTER TYPE category ADD VALUE 'Professional';
ALTER TYPE category ADD VALUE 'HomeSkills';

-- Vertical-specific attributes
CREATE TABLE vertical_attributes (
    id UUID PRIMARY KEY,
    tutor_profile_id UUID REFERENCES tutor_profiles(id),
    vertical VARCHAR(50) NOT NULL,
    attribute_key VARCHAR(100) NOT NULL,
    attribute_value TEXT,
    verified BOOLEAN DEFAULT FALSE,
    verified_at TIMESTAMP,
    UNIQUE(tutor_profile_id, attribute_key)
);

-- Examples:
-- ('Driving', 'adi_badge_number', 'ADI123456', true)
-- ('Fitness', 'reps_number', 'R1234567', true)
-- ('Fitness', 'insurance_provider', 'Markel', false)
-- ('Driving', 'transmission_type', 'Manual,Automatic', false)
-- ('Driving', 'car_model', 'Ford Fiesta 2023', false)
```
